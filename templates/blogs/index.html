Chapter 3

1,000,000 dataset

Data is in format:

  • "rating.dat" : UserID::MovieID::Rating::Timestamp
  • "users.dat" : UserID::Gender::Age::Occupation::Zip-code
  • "movies.dat" : MovieID::Title::Genres
In [9]: 
if True:
	user_cols = ['UserID', 'Gender', 'Age', 'Occupation', 'Zip-code']
	users = pd.read_table('data/users.dat', sep='::', header=None, names=user_cols)

	rating_cols = ['UserID', 'MovieID', 'Rating', 'Timestamp']
	ratings = pd.read_table('data/ratings.dat', sep='::', header=None, names=rating_cols)

	movie_cols = ['MovieID', 'Title', 'Genres']
	movies = pd.read_table('data/movies.dat', sep='::', header=None, names=movie_cols)
	# merge 3 data frame into 1 for easily manipulate
	data = pd.merge(pd.merge(users, ratings), movies)
